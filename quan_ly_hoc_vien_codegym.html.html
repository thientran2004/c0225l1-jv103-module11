<h1>Quản Lý Học Viên CODEGYM</h1>
<button onclick="addStudent()">Thêm Mới Học Viên</button>
<button onclick="displayStudents()">Hiển Thị Học Viên</button>

<table id="studentTable">
  <thead>
  <tr>
    <th>Mã Học Viên</th>
    <th>Tên</th>
    <th>Lớp</th>
    <th>Email</th>
    <th>Ngày Sinh</th>
    <th>Module</th>
    <th>Chỉnh Sửa</th>
  </tr>
  </thead>
  <tbody>
  </tbody>
</table>

<script>
  let students = [];

  function displayStudents() {
    const tableBody = document.querySelector('#studentTable tbody');
    tableBody.innerHTML = '';
    students.forEach((student, index) => {
      const row = tableBody.insertRow();
      row.innerHTML = `
                <td>${student.id}</td> 
                <td>${student.name}</td> 
                <td>${student.class}</td> 
                <td>${student.email}</td> 
                <td>${student.birthDate}</td> 
                <td>${student.module}</td> 
                <td> 
                    <button onclick="editStudent(${index})">Sửa</button> 
                    <button onclick="deleteStudent(${index})">Xóa</button> 
                </td> 
            `;
    });
  }

  function addStudent() {
    const id = prompt("Nhập mã học viên (HV-XXXX):");
    const name = prompt("Nhập tên học viên:");
    const className = prompt("Nhập lớp:");
    const email = prompt("Nhập email:");
    const birthDate = prompt("Nhập ngày sinh (dd/mm/yyyy):");
    const module = prompt("Nhập module:");

    const student = { id, name, class: className, email, birthDate, module };
    students.push(student);
    displayStudents();
  }

  function editStudent(index) {
    const updatedName = prompt("Nhập tên học viên mới:", students[index].name);
    students[index].id = prompt("Nhập id mới:", students[index].class);
    students[index].name = updatedName;
    students[index].class = prompt("Nhập lớp mới:", students[index].class);
    students[index].email = prompt("Nhập email mới:", students[index].email);
    students[index].birthDate = prompt("Nhập ngày sinh mới (dd/mm/yyyy):", students[index].birthDate);
    students[index].module = prompt("Nhập module mới:", students[index].module);
    displayStudents();
  }

  function deleteStudent(index) {
    if (confirm(`Bạn có chắc chắn muốn xóa học viên với mã ${students[index].id}?`)) {
      students.splice(index, 1);
      displayStudents();
    }
  }
</script>
